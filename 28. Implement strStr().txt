class Solution(object):
    def strStr(self, haystack, needle):
        """
        :type haystack: str
        :type needle: str
        :rtype: int
        """
        # linear comparing
        # time complexity: O(n)
        for i in range(len(haystack)-len(needle)+1):
            if self.check_same(haystack[i:i+len(needle)], needle):
                return i
        return -1
