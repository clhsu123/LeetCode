	path = []
        ans = []
        
        def helper(path, n, ans):
            if sum(path) == n:
                ans.append(list(path))
                return
            elif sum(path) > n:
                return
            else:
                for i in range(1, 3):
                    path.append(i)
                    helper(path, n, ans)
                    path.pop()
        helper(path, n, ans)
        return ans
